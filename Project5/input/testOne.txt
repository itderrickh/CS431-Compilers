BEGIN
    VOID main() {
        y : INT;
        y := 20;
        x : INT;
        x := 30;
        h : INT;
        h := 40;
        z : INT;
        z := 2;
        SWITCH(z) {
            CASE (1):
                PUT(y);
            CASE (2):
                PUT(z);
            CASE (3):
                PUT(h);
                BREAK;
            DEFAULT:
                PUT(x);
        }
    }
END
